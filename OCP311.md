


```bash
SUBSCRIPTION='Red Hat OpenShift, Standard Support (10 Cores, NFR, Partner Only)'

POOL_ID=$( subscription-manager list --available --matches "$SUBSCRIPTION" | grep "Pool ID:" | cut -d: -f2 )

subscription-manager attach --pool=$POOL_ID
```
